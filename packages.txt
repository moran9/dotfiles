# Packages required by this dotfiles repo. Installed by apply.sh via yay.
# Comments and blank lines are ignored.

# ── Compositor ─────────────────────────────────────────────────────
hyprland
hyprpaper
hyprshot
hyprpolkitagent
hypridle             # idle daemon (auto-lock, DPMS, suspend)
hyprlock             # lock screen

xdg-desktop-portal-hyprland
xdg-desktop-portal-gtk     # native file-chooser portal for Electron/Chromium
xsettingsd                 # broadcasts gsettings → XSettings for XWayland apps

# ── Bar / launcher / notifications ─────────────────────────────────
waybar
wofi
wlogout              # logout / power menu launched from waybar's power pill
mako
copyq                # GUI clipboard manager: history, images, tray

# ── Terminal & shell ───────────────────────────────────────────────
kitty
zsh

# ── Screenshots / clipboard ────────────────────────────────────────
grimblast-git
grim
slurp
satty                # screenshot annotation editor (Wayland-native)
wl-clipboard

# ── Media / system control ─────────────────────────────────────────
playerctl
brightnessctl
pavucontrol
pulsemixer           # minimalist TUI mixer used by the waybar audio popup
nm-connection-editor # GUI for NM profiles (right-click on waybar network pill)

bluez                # bluetooth stack

blueman              # full GTK manager (right-click on bt pill)

papirus-icon-theme

# ── Theming ────────────────────────────────────────────────────────
# GTK is the primary theme target. For Qt, we use a minimal setup:
# qt6ct/qt5ct as the platform-theme bridge, Fusion as the widget style,
# and a static Catppuccin palette file. This is enough to make Qt apps
# (CopyQ, pavucontrol, etc.) render dark without re-opening the
# Dolphin/Kvantum/hyprqt6engine saga (which went nowhere — KDE
# Frameworks apps remain unthemed; just don't use them).
catppuccin-gtk-theme-mocha            # Catppuccin GTK3/4 theme (AUR) — primary theme
qt5ct                                 # Qt5 platform theme bridge
qt6ct                                 # Qt6 platform theme bridge
sddm-catppuccin-git                   # Catppuccin login screen (AUR)
qt5-graphicaleffects                  # required by SDDM Qt5 themes
qt5-svg                               # required by SDDM Qt5 themes
qt5-quickcontrols2                    # required by SDDM Qt5 themes

# ── Utilities ──────────────────────────────────────────────────────
hdrop-git            # dropdown terminal helper (AUR; declared makedeps are incomplete — see BUILD_DEPS in apply.sh)
nautilus             # GTK file manager (omarchy/ML4W ship this; reads catppuccin-gtk natively)
nautilus-python      # nautilus extension support
nwg-displays         # GUI monitor configurator (helps build monitors.conf)
imagemagick          # used by apply.sh to generate fallback wallpaper
python               # for the claude_usage.py waybar widget
